function Start()
	LoadGameCtrl()
	ViewManager:Open(View.MainView)
end

function LoadGameCtrl()
	local require_list = require("require_list")
	for _, v in ipairs(require_list) do
		require(v)
	end

	Utils.New()
	Clock.New()
	EventManager.New()
	ViewManager.New()
	SoundManager.New()
	SceneManager.New()
	UpdateManager.New()

	PlayCtrl.New()
	LoadingCtrl.New()
	MainViewCtrl.New()
	RoleExpCtrl.New()
	TurntableCtrl.New()
end

function HandleEvent(event)
	EventManager:Fire(event.eventID, event)
end

function Update()
	UpdateManager:Update()
end

function FixedUpdate()
	UpdateManager:FixedUpdate()
end

function LateUpdate()
	UpdateManager:LateUpdate()
end
